class Solution {
    public boolean canJump(int[] nums) {
      int steps=nums[0];
      for(int i=0;i<=nums.length-1;i++)
      {
        steps--;
        
        if((nums[i]<0))  return false;
           
        if(nums[i]+steps>nums.length-1) return  false;
        
        if(nums[i]>steps)
        {
            steps=nums[i];
        }
        
        if(i==nums.length-1)   return true;
          
      }
    return true;
    }
}
